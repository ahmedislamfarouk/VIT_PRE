\begin{abstract}
Vision Transformers (ViTs) have revolutionized computer vision by demonstrating superior scalability and representational power compared to convolutional architectures. This paper presents a rigorous comparative analysis of three prominent Meta AI Vision Transformer-based models: Segment Anything Model 2 (SAM 2), Data-efficient Image Transformer (DeiT), and DINOv3. We define their core architectures, derive complete mathematical formulations including multi-head self-attention, positional encodings, distillation objectives, self-supervised contrastive losses, and prompt-based segmentation mechanisms, and conduct a deep analysis across architectural design, training paradigms, advantages, limitations, and computational complexity. Finally, we propose three novel research-level enhancements integrating features across these models. Our analysis underscores the complementary strengths of these models and charts a path toward hybrid foundation models for universal vision tasks.
\end{abstract}

\section{Introduction}
The advent of the Vision Transformer (ViT) marked a paradigm shift in computer vision, replacing inductive biases of convolutions with scalable self-attention mechanisms. Subsequent Meta AI models have extended ViTs in distinct directions: SAM 2 for promptable segmentation in images and videos, DeiT for data-efficient supervised classification, and DINOv3 for large-scale self-supervised representation learning. 

This work provides the first comprehensive, equation-level comparison of these three models. We dissect their architectures, derive mathematical formulations, perform a multi-dimensional comparative analysis, and propose novel enhancements that synthesize insights across models.

\section{Model Architectures}

\subsection{SAM 2 Vision Transformer}
SAM 2 employs a transformer-based image encoder (derived from ViT with hierarchical adaptations akin to Hiera) for feature extraction. The encoder processes inputs into patch tokens. A streaming memory module maintains object-specific embeddings across frames via cross-attention. The mask decoder is a lightweight transformer that fuses image embeddings with prompt tokens (points, boxes, masks) and memory tokens to predict binary masks.

\subsection{DeiT Vision Transformer}
DeiT retains the canonical ViT architecture but introduces a distillation token alongside the class token. For an input image, it is split into patches, linearly projected, and augmented with positional encodings and the two special tokens. The distillation token learns to mimic a teacher (typically a CNN) via attention interactions, enabling data-efficient supervised training on ImageNet-1k.

\subsection{DINOv3 Vision Transformer}
DINOv3 scales ViT to unprecedented sizes (up to 7B parameters) with architectural refinements: axial Rotary Position Embeddings (RoPE), 4 register tokens, SwiGLU FFNs, and dedicated projection heads for global (DINO) and patch-level (iBOT) objectives. Training uses multi-crop views. DINOv3 introduces a Gram anchoring phase to stabilize dense features after extensive training.

\section{Mathematical Formulation}
All models share the core MSA mechanism from ViT. Let $\mathbf{X} \in \mathbb{R}^{N \times D}$ be the input token sequence:
\begin{equation}
\mathbf{Q} = \mathbf{X}\mathbf{W}^Q, \quad \mathbf{K} = \mathbf{X}\mathbf{W}^K, \quad \mathbf{V} = \mathbf{X}\mathbf{W}^V
\end{equation}
\begin{equation}
\text{MSA}(\mathbf{X}) = \text{Concat}(\text{head}_1, \dots, \text{head}_H)\mathbf{W}^O
\end{equation}
\begin{equation}
\text{head}_i = \text{Softmax}\left(\frac{\mathbf{Q}_i\mathbf{K}_i^\top}{\sqrt{d}}\right)\mathbf{V}_i
\end{equation}

\subsection{SAM 2: Prompt-Based Segmentation}
Prompt tokens $\mathbf{P}$ are added to image features $\mathbf{F}_{\text{img}}$ via cross-attention:
\begin{equation}
\mathbf{F}_{\text{dec}} = \text{CrossAttn}(\mathbf{P}, \mathbf{F}_{\text{img}} + \mathbf{M}_{\text{mem}})
\end{equation}
Streaming memory update: $\mathbf{M}^{(t)}_{\text{mem}} = \alpha \mathbf{M}^{(t-1)}_{\text{mem}} + (1 - \alpha)\mathbf{F}^{(t)}_{\text{img}}$.

\subsection{DeiT: Knowledge Distillation}
DeiT augments input with class token $\mathbf{t}_{\text{cls}}$ and distillation token $\mathbf{t}_{\text{dist}}$:
\begin{equation}
\mathbf{X}' = [\mathbf{X}_{\text{patch}}; \mathbf{t}_{\text{cls}}; \mathbf{t}_{\text{dist}}]
\end{equation}
Hard distillation loss:
\begin{equation}
\mathcal{L}_{\text{hard}} = \frac{1}{2}\mathcal{L}_{\text{CE}}(\mathbf{z}_{\text{cls}}, y) + \frac{1}{2}\mathcal{L}_{\text{CE}}(\mathbf{z}_{\text{dist}}, y_t)
\end{equation}

\subsection{DINOv3: Self-Supervised Objectives}
DINO loss (global):
\begin{equation}
\mathcal{L}_{\text{DINO}} = - \sum \mathbf{p}_t \log \mathbf{p}_s
\end{equation}
Gram anchoring (key innovation):
\begin{equation}
\mathcal{L}_{\text{Gram}} = \|\mathbf{X}_s\mathbf{X}_s^\top - \mathbf{X}_g\mathbf{X}_g^\top\|^2_F
\end{equation}
Full pre-training loss:
\begin{equation}
\mathcal{L}_{\text{Pre}} = \mathcal{L}_{\text{DINO}} + \mathcal{L}_{\text{iBOT}} + 0.1\mathcal{L}_{\text{DKoleo}} + w_{\text{Gram}}\mathcal{L}_{\text{Gram}}
\end{equation}

\section{Proposed Enhancements}
\textbf{Enhancement 1: DINOv3-Augmented SAM 2 Encoder.} Integrate DINOv3's frozen dense features as additional input to SAM 2 via cross-attention:
\begin{equation}
\mathbf{F}_{\text{aug}} = \mathbf{F}_{\text{img}} + \text{CrossAttn}(\mathbf{F}_{\text{img}}, \mathbf{F}_{\text{DINO}})
\end{equation}

\textbf{Enhancement 2: Dynamic Register + Gram Distillation for DeiT.} Extend DeiT's distillation token with DINOv3-style registers and Gram anchoring during supervised training.

\textbf{Enhancement 3: Prompt-Conditioned Self-Supervised Objective.} Unify prompt encoding into DINOv3's teacher-student loop:
\begin{equation}
\mathcal{L}_{\text{prompt-DINO}} = \mathcal{L}_{\text{DINO}} + \lambda\mathcal{L}_{\text{prompt-cross}}
\end{equation}

\section{Conclusion}
Our analysis reveals complementary strengths: SAM 2 for interactive tasks, DeiT for efficiency, and DINOv3 for universality. The proposed enhancements offer concrete paths to hybrid ViTs with superior performance.
